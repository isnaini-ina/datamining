def diss(x,y):
  a = 4
  b = 0
  for col in ["Country", "Confirmed Cases", "Deaths",	"New Cases",	"Country Population"]:
    if p[col][x] == p[col][y]:
      b += 1
    return (a-b)/a

print(diss(1,1))

  result = 0
  for col in [hasil1, hasil2]:
    result = math.pow(abs(col[a] -col[b]), c)
  return result**(1/c)
